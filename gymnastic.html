console.log("🏆 DOLPHINS vs KOALAS SCORING CHALLENGE 🏆\n"); // Helper function to calculate average function calculateAverage(scores) { const sum = scores.reduce((total, score) => total + score, 0); return sum / scores.length; } // =========================================== // PART 1: Basic Trophy Rules // =========================================== console.log("=== PART 1: Basic Trophy Rules ==="); // Test Data 1 const dolphins1 = [96, 108, 89]; const koalas1 = [88, 91, 110]; console.log(`Dolphins scores: ${dolphins1.join(', ')}`); console.log(`Koalas scores: ${koalas1.join(', ')}`); // Calculate averages const dolphinsAvg1 = calculateAverage(dolphins1); const koalasAvg1 = calculateAverage(koalas1); console.log(`Dolphins average: ${dolphinsAvg1.toFixed(1)}`); console.log(`Koalas average: ${koalasAvg1.toFixed(1)}`); // Determine winner - basic rules if (dolphinsAvg1 > koalasAvg1) { console.log("Dolphins win the trophy 🏆"); } else if (koalasAvg1 > dolphinsAvg1) { console.log("Koalas win the trophy 🏆"); } else { console.log("Both teams win the trophy 🏆"); } // =========================================== // PART 2: Minimum Score Requirement // =========================================== console.log("\n=== PART 2: Minimum Score Requirement (100 points) ===");
